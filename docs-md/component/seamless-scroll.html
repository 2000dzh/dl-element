支持数据项动态变化 ​

效果如下

<script setup lang="ts">
import { ref } from "vue";
const list = ref(
	Array.from({ length: 5 }, (_, index) => ({
		name: `测试${index + 1}`
	}))
);

const add = () => {
	const len = list.value.length;
	list.value.push({
		name: `测试${len + 1}`
	});
};

const subtract = () => {
	list.value.pop();
};

const handler = () => {
	list.value = Array.from({ length: 5 }, (_, index) => ({
		name: `测试${index + 1}`
	}));
};
</script>

<template>
  <div>
    <ClientOnly>
      <DlSeamlessScroll class="the-demo-list">
        <div
          v-for="obj of list"
          :key="obj.name"
          class="the-demo-list__item"
        >
          {{ obj.name }}
        </div>
      </DlSeamlessScroll>
    </ClientOnly>

    <button @click="add">
      增加元素
    </button>
    <button @click="subtract">
      减少元素
    </button>

    <button @click="handler">
      重新赋值
    </button>
  </div>
</template>
